// Firebase Storage Security Rules for Profile Pictures
// Copy and paste these rules into Firebase Console > Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Profile Pictures Rules
    // Path structure: profile_pictures/{userId}/profile.{extension}
    match /profile_pictures/{userId}/{allPaths=**} {
      // Anyone can read profile pictures (needed for displaying in chats)
      allow read: if true;
      
      // Only authenticated users can write to their own folder
      allow write, delete: if request.auth != null 
                           && request.auth.uid == userId;
    }
    
    // Additional validation for profile picture uploads
    match /profile_pictures/{userId}/profile.{extension} {
      allow write: if request.auth != null 
                   && request.auth.uid == userId
                   && request.resource.size < 5 * 1024 * 1024  // Max 5MB
                   && request.resource.contentType.matches('image/.*');  // Only images
    }
  }
}

// 3. Click "Storage" in the left sidebar
// 4. If not enabled, click "Get Started" to enable Storage
// 5. Click on the "Rules" tab
// 6. Replace the existing rules with the rules above
// 7. Click "Publish"

// - Maximum file size is 5MB (adjustable)
// - Only image files are allowed
// - All users can read profile pictures (needed for app functionality)
// - Files are automatically deleted when user deletes their profile picture
